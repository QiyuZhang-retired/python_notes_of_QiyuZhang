# T02811:熄灯问题
**核心算法：枚举法**
```python
#设置按下灯熄灭的函数方便调用
def pull(i,j):
    for di,dj in ((-1,0),(0,0),(1,0),(0,-1),(0,1)):
        if 0<=i+di<=4 and 0<=j+dj<=5:
            if light_copy[i+di][j+dj]==0:
                light_copy[i+di][j+dj]=1
            else:
                light_copy[i+di][j+dj]=0

    return

light=[]
for _ in range(5):
    light.append(list(map(int,input().split())))
output=[]

for n in range(64):
    #注意矩阵行的浅拷贝
    light_copy=[]
    for row in light:
        light_copy.append(row[:])
    output=[[0]*6 for _ in range(5)]

    #枚举每一个第一行的选取方式
    k=0
    while n>0:
        if n%2==1:
            output[0][k]=1
            pull(0,k)

        n//=2
        k+=1

    #根据题意修正下面的行
    for i in range(4):
        for j in range(6):
            if light_copy[i][j]==1:
                output[i+1][j]=1
                pull(i+1,j)

    #验证此第一行选取方式下，操作完后最后一行是不是符合题意的
    if light_copy[4]==[0,0,0,0,0,0]:
        break

for row in output:
    print(' '.join(list(map(str,row))))
```
