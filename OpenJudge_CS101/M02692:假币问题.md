# M02692:假币问题
**核心算法：穷举法**  
数据量很小可以一一枚举，使用eval方便进行判断，不重复写代码而是让代码自己写代码
```python
coins={'A':0,'B':0,'C':0,'D':0,'E':0,'F':0,
       'G':0,'H':0,'I':0,'J':0,'K':0,'L':0}
weight={1:'heavy',-1:'light'}

n=int(input())

for _ in range(n):
    compares=[]

    for _ in range(3):
        origin_compare=input().split()
        compare=''

        compare+=f'coins[\'{origin_compare[0][0]}\']'
        for i in range(1,4):
            compare+=f'+coins[\'{origin_compare[0][i]}\']'

        if origin_compare[2]=='even':
            compare+='=='
        elif origin_compare[2]=='up':
            compare+='>'
        else:
            compare+='<'

        compare+=f'coins[\'{origin_compare[1][0]}\']'
        for i in range(1,4):
            compare+=f'+coins[\'{origin_compare[1][i]}\']'

        compares.append(compare)

    for letter in ('A','B','C','D','E','F','G','H','I','J','K','L'):
        for i in (-1,1):
            coins[letter]=i

            if eval(compares[0]) and eval(compares[1]) and eval(compares[2]):
                print(f'{letter} is the counterfeit coin and it is {weight[i]}.')

            coins[letter]=0
```
