# E02724:生日相同
```python
n=int(input())
birthdays=[]
for _ in range(n):
    ID,m,d=input().split()
    birthdays.append((int(m),int(d),ID))

dict_birthdays={}
for m,d,ID in birthdays:
    key=(m,d)
    if key not in dict_birthdays:
        dict_birthdays[key]=[]
    dict_birthdays[key].append(ID)

sorted_dict_birthdays=sorted(dict_birthdays)
for date in sorted_dict_birthdays:
    if len(dict_birthdays[date])>1:
        month,day=date
        print(f'{month} {day} {" ".join(dict_birthdays[date])}')
```
