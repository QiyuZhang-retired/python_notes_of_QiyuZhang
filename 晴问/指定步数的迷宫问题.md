# 指定步数的迷宫问题
**核心算法：深度优先搜索**  
递归判定是否能成功，一旦成功就返回True结束循环
```python
def dfs(x,y,maze,steps):
    for dx,dy in ((-1,0),(1,0),(0,-1),(0,1)):
        nx=x+dx
        ny=y+dy

        if maze[nx][ny]=='end' and steps==k:
            return True

        if maze[nx][ny]==0:
            maze[x][y]=1
            if dfs(nx,ny,maze,steps+1):
                return True
            maze[x][y]=0

    return False


n,m,k=map(int,input().split())
maze=[]
maze.append([1]*(m+2))
for _ in range(n):
    maze.append([1]+list(map(int,input().split()))+[1])
maze.append([1]*(m+2))
maze[1][1]='start'
maze[n][m]='end'

if dfs(1,1,maze,1):
    print('Yes')
else:
    print('No')
```
