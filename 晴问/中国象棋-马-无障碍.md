# 中国象棋-马-无障碍
**算法核心：广度优先搜索**  
多终点迷宫问题的改版，修改移动方式即可
```python
from collections import deque

def bfs(board,x,y):
    in_queue=set()
    q=deque()
    in_queue.add((x+1,y+1))
    q.append((0,x+1,y+1))

    while q:
        front=q.popleft()
        board[front[1]][front[2]]=front[0]

        for dx,dy in ((-2,-1),(-2,1),(-1,-2),(-1,2),(1,-2),(1,2),(2,-1),(2,1)):
            nx=front[1]+dx
            ny=front[2]+dy
            if (nx,ny) not in in_queue and board[nx][ny]!=float('-inf'):
                in_queue.add((nx,ny))
                q.append((front[0]+1,nx,ny))

    return

n,m,x,y=map(int,input().split())
board=[[float('-inf')]*(m+4),[float('-inf')]*(m+4)]
for _ in range(n):
    board.append([float('-inf')]*2+[-1]*m+[float('-inf')]*2)
board+=[[float('-inf')]*(m+4),[float('-inf')]*(m+4)]
bfs(board,x,y)

for row in board[2:-2]:
    print(' '.join(list(map(str,row[2:-2]))))
```
