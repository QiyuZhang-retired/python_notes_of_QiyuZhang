# 数字操作
**核心算法：广度优先搜索**  
1）使用deque来快速进行入队出队  
2）使用集合记录已入队的元素，查询更快  
```python
from collections import deque

def bfs(n):
    in_queue=set()
    q=deque()

    #第一个元素入队
    in_queue.add(1)
    q.append((0,1))

    #逐层遍历，按层数队尾入队，队首出队，in_queue记录是否已入队
    while q:
        step,front=q.popleft()
        if front==n:
            return step

        if front*2<=n and front*2 not in in_queue:
            in_queue.add(front*2)
            q.append((step+1,front*2))
        if front+1<=n and front+1 not in in_queue:
            in_queue.add(front+1)
            q.append((step+1,front+1))

n=int(input())
print(bfs(n))
```
