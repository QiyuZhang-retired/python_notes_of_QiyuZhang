# 迷宫可行路径数
**算法核心：深度优先搜素**  
1）利用global来声称全局变量，利于控制输出  
2）递归dfs遍历每一条路径
```python
def dfs(x,y,maze):
    #global调用全局变量方便输出
    global cnt

    for dx,dy in ((-1,0),(1,0),(0,-1),(0,1)):
        nx=x+dx
        ny=y+dy

        if maze[nx][ny]=='end':
            cnt+=1
            continue

        if maze[nx][ny]==0:
            maze[x][y]=1
            dfs(nx,ny,maze)
            maze[x][y]=0

    return


n,m=map(int,input().split())
maze=[]
#加保护圈避免越界
maze.append([1]*(m+2))
for _ in range(n):
    maze.append([1]+list(map(int,input().split()))+[1])
maze.append([1]*(m+2))

maze[1][1]='start'
maze[n][m]='end'

cnt=0
dfs(1,1,maze)
print(cnt)
```
