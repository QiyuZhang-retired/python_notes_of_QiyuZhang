# 字符迷宫
**算法核心：广度优先搜索**  
将01迷宫变成字符迷宫，思路不变
```python
from collections import deque

def bfs(maze,xs,ys,xt,yt):
    in_queue=set()
    q=deque()
    in_queue.add((xs,ys))
    q.append((0,xs,ys))

    while q:
        front=q.popleft()
        if front[1:]==(xt,yt):
            return front[0]

        for dx,dy in ((-1,0),(1,0),(0,-1),(0,1)):
            nx=front[1]+dx
            ny=front[2]+dy
            if (nx,ny) not in in_queue and (maze[nx][ny]=='.' or maze[nx][ny]=='T'):
                in_queue.add((nx,ny))
                q.append((front[0]+1,nx,ny))

    return -1

n,m=map(int,input().split())
maze=[['*']*(m+2)]
for _ in range(n):
    maze.append(['*']+[c for c in input()]+['*'])
maze+=[['*']*(m+2)]

xs,ys,xt,yt=0,0,0,0
for i in range(1,n+1):
    for j in range(1,m+1):
        if maze[i][j]=='S':
            xs,ys=i,j
        if maze[i][j]=='T':
            xt,yt=i,j

print(bfs(maze,xs,ys,xt,yt))
```
