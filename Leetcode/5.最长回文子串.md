# 5.最长回文子串
**核心算法：动态规划**   
1）用dp表示第i到第j位的子串是否是回文子串  
2）进行上三角填充，保证填充下一个前所需的状态已经填好  
```python
class Solution:
    def longestPalindrome(self, s: str) -> str:

        n=len(s)
        if n==1:
            return s

        #设置状态转移函数与状态标记变量
        dp=[[False]*n for _ in range(n)]
        start=0
        lenth=1

        #进行状态转移
        for i in range(n-1,-1,-1):
            for j in range(i,n):
                if i==j:
                    dp[i][j]=True
                elif s[i]==s[j]:
                    if j-i==1:
                        dp[i][j]=1
                    else:    
                        dp[i][j]=dp[i+1][j-1]

                if lenth<j-i+1 and dp[i][j]:
                    lenth=j-i+1
                    start=i

        return s[start:start+lenth]
```
